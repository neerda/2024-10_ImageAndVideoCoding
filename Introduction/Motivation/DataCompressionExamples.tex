

\begin{frame}{Practical Data Compression Examples}
\vspace{-.5ex}%
\uncover<1->{\STRUC{File Compression}
  \bit
  \item Compress large document archives
  \item[\iarrow] WinZip, WinRar, gzip, lzip, bzip2, 7zip, ...
  \eit
}
\uncover<2->{\STRUC{Audio Compression}
  \bit
  \item Compress music or audio books for storage on mobile devices
  \item[\iarrow] FLAC, MP3, AAC, ...
  \eit
}
\uncover<3->{\STRUC{Image Compression}
  \bit
  \item Compress images for storage and distribution
  \item[\iarrow] Raw camera formats, PNG, JPEG, JPEG-2000, JPEG-XR, H.265~| HEVC, ...
  \eit
}
\uncover<4->{\STRUC{Video Compression}
  \bit
  \item Compress video for streaming, broadcast, or storage
  \item[\iarrow] MPEG-2, H.264~|~AVC, VP9, H.265~|~HEVC, ...
  \eit
}
\end{frame}
