
\begin{frame}{Motivation for Data Compression (or Source Coding)}
\uncover<2->{\STRUC{Application of Data Compression}
\bit
\item Efficient transmission or storage of data
\item[\iarrow] {\bf Use less throughput for given data}
\item[\iarrow] {\bf Transmit more data for given throughput}
\eit}

\medskip
\uncover<3->{\STRUC{Data Compression: Enabling Technology}
\bit
\item Enables new applications
\item Makes applications economically feasible
\item<4->[\clg{\iarrow}]\smallskip \clg{\bf Well-known examples:}
  \bit\relsize{1}
  \item[\clg{\icircle}] Distribution of digital images
  \item[\clg{\icircle}] Distribution of digital audio
  \item[\clg{\icircle}] Digitial television (DVB-T, DVB-T2, DVB-S, DVB-S2)
  \item[\clg{\icircle}] Internet video streaming (YouTube, Netflix, Amazon, ...)
  \eit
\eit}
\end{frame}
