
\begin{frame}{Practical Source Coding Problem}
%\vspace{-1.0ex}
\STRUC{Characteristics of Source Codecs}
\bit
\item Bit rate: Throughput of the communication channel
\item Quality:   Fidelity of the reconstructed signal
\item Delay:      Start-up latency, end-to-end delay
\item Complexity: Computation, memory, memory access
\eit






\bigskip

  \begin{center}\begin{minipage}{0.8\linewidth}\vspace{-3.0ex}
  \basblk{}
  \ALERT{\begin{center}

   achieve an optimal trade-off between bit rate and reconstruction quality
   for the transmission problem in the targeted application
  \end{center}}

  \end{minipage}\end{center}
